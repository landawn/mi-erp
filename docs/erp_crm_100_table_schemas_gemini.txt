### ERP/CRM Database Schema Structure

#### Global Standards Used:

  * **Engine:** InnoDB (for ACID compliance).
  * **Keys:** `BIGINT` for primary keys (scalability).

### Phase 1: Core, Identity & Common Data

These tables support the entire ecosystem.

```sql
-- 1. Users (System Access)
CREATE TABLE users (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(255) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    first_name VARCHAR(100),
    last_name VARCHAR(100),
    is_active BOOLEAN DEFAULT TRUE,
    last_login DATETIME,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 2. Roles (RBAC)
CREATE TABLE roles (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. Currencies (Multi-currency support)
CREATE TABLE currencies (
    code CHAR(3) PRIMARY KEY, -- USD, EUR
    name VARCHAR(50),
    symbol VARCHAR(5),
    exchange_rate DECIMAL(10, 4) DEFAULT 1.0000,
    is_base BOOLEAN DEFAULT FALSE
);

-- 4. Countries & Regions
CREATE TABLE countries (
    iso_code CHAR(2) PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

-- 5. Addresses (Polymorphic or shared)
CREATE TABLE addresses (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    parent_type ENUM('account', 'contact', 'warehouse', 'employee'),
    parent_id BIGINT UNSIGNED NOT NULL,
    street_line_1 VARCHAR(255),
    street_line_2 VARCHAR(255),
    city VARCHAR(100),
    state VARCHAR(100),
    postal_code VARCHAR(20),
    country_code CHAR(2),
    address_type ENUM('billing', 'shipping', 'home', 'office'),
    FOREIGN KEY (country_code) REFERENCES countries(iso_code),
    INDEX (parent_type, parent_id)
);
```


### Phase 3: Inventory & Products (Supply Chain)

Managing what you sell and where it is.

```sql
-- 10. Product Categories
CREATE TABLE product_categories (

    name VARCHAR(100) NOT NULL,

    FOREIGN KEY (parent_id) REFERENCES product_categories(id)

CREATE TABLE products (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    sku VARCHAR(50) NOT NULL UNIQUE,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    category_id BIGINT UNSIGNED,
    uom VARCHAR(20) DEFAULT 'pcs', -- Unit of Measure

    sales_price DECIMAL(15, 4),
    is_stockable BOOLEAN DEFAULT TRUE,
    FOREIGN KEY (category_id) REFERENCES product_categories(id)
);

-- 12. Warehouses
CREATE TABLE warehouses (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    code VARCHAR(20) UNIQUE,
    location_address_id BIGINT UNSIGNED
);

-- 13. Inventory Stock (Current Levels)
CREATE TABLE inventory_stock (
    product_id BIGINT UNSIGNED,
    warehouse_id BIGINT UNSIGNED,
    quantity_on_hand DECIMAL(15, 4) DEFAULT 0,
    quantity_reserved DECIMAL(15, 4) DEFAULT 0,
    reorder_point DECIMAL(15, 4) DEFAULT 10,
    PRIMARY KEY (product_id, warehouse_id),
    FOREIGN KEY (product_id) REFERENCES products(id),
    FOREIGN KEY (warehouse_id) REFERENCES warehouses(id)
);

-- 14. Inventory Movements (Audit Trail)
CREATE TABLE inventory_movements (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    product_id BIGINT UNSIGNED NOT NULL,
    warehouse_id BIGINT UNSIGNED NOT NULL,
    movement_type ENUM('in', 'out', 'transfer', 'adjustment'),

    reference_id BIGINT UNSIGNED, -- Link to Order/PO ID
    reference_type VARCHAR(50),   -- 'sales_order', 'purchase_order'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

```sql
-- 18. Chart of Accounts (COA)

CREATE TABLE invoices (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    invoice_number VARCHAR(50) NOT NULL UNIQUE,
    sales_order_id BIGINT UNSIGNED,
    customer_id BIGINT UNSIGNED,
    issue_date DATE,
    due_date DATE,
    total_amount DECIMAL(15, 2),
    status ENUM('draft', 'posted', 'paid', 'overdue'),
    FOREIGN KEY (sales_order_id) REFERENCES sales_orders(id)
);

-- 21. Payments
CREATE TABLE payments (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    invoice_id BIGINT UNSIGNED,
    payment_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    amount DECIMAL(15, 2) NOT NULL,
    payment_method ENUM('credit_card', 'bank_transfer', 'cash', 'check'),
    transaction_reference VARCHAR(100),
    FOREIGN KEY (invoice_id) REFERENCES invoices(id)
);

-- 22. Journal Entries (The Ledger Header)
CREATE TABLE journal_entries (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    entry_number VARCHAR(50) UNIQUE,
    reference VARCHAR(100),
    date DATE NOT NULL,
    description TEXT,
    posted BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 23. Journal Entry Lines (Debits/Credits)
CREATE TABLE journal_entry_lines (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    journal_entry_id BIGINT UNSIGNED NOT NULL,
    account_id BIGINT UNSIGNED NOT NULL, -- Link to COA
    debit DECIMAL(15, 2) DEFAULT 0.00,
    credit DECIMAL(15, 2) DEFAULT 0.00,
    FOREIGN KEY (journal_entry_id) REFERENCES journal_entries(id),
    FOREIGN KEY (account_id) REFERENCES chart_of_accounts(id)
);
```
